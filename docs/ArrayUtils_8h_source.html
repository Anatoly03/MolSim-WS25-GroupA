91template <class Container>
92[[nodiscard]] std::string to_string(const Container &container, const std::string &delimiter = ", ",
93 const std::array<std::string, 2> &surround = {"[", "]"}) {
94 auto iter = std::cbegin(container);
95 const auto end = std::cend(container);
96 if (iter == end) {
97 return surround[0] + surround[1];
98 }
99 std::ostringstream strStream;
100 strStream << surround[0] << *iter;
101 for (++iter; iter != end; ++iter) {
102 strStream << delimiter << *iter;
103 }
104 strStream << surround[1];
105 return strStream.str();
106}

121template <class Container, class F>
122inline Container elementWisePairOp(const Container &lhs, const Container &rhs, F binaryFunction) {
123 Container ret = lhs;
124 auto retIter = std::begin(ret);
125 auto lhsIter = std::cbegin(lhs);
126 const auto lhsEnd = std::cend(lhs);
127 auto rhsIter = std::cbegin(rhs);
128 const auto rhsEnd = std::cend(rhs);
129
130 for (; lhsIter != lhsEnd and rhsIter != rhsEnd; ++lhsIter, ++rhsIter, ++retIter) {
131 *retIter = binaryFunction(*lhsIter, *rhsIter);
132 }
133
134 return ret;
135}
